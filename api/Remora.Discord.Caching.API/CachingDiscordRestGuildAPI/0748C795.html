public Task<Result<IGuild>> ModifyGuildAsync(Snowflake guildID, Optional<string> name = null, Optional<VerificationLevel?> verificationLevel = null, Optional<MessageNotificationLevel?> defaultMessageNotifications = null, Optional<ExplicitContentFilterLevel?> explicitContentFilter = null, Optional<Snowflake?> afkChannelID = null, Optional<TimeSpan> afkTimeout = null, Optional<Stream?> icon = null, Optional<Snowflake> ownerID = null, Optional<Stream?> splash = null, Optional<Stream?> discoverySplash = null, Optional<Stream?> banner = null, Optional<Snowflake?> systemChannelID = null, Optional<SystemChannelFlags> systemChannelFlags = null, Optional<Snowflake?> rulesChannelID = null, Optional<Snowflake?> publicUpdatesChannelID = null, Optional<string?> preferredLocale = null, Optional<IReadOnlyList<GuildFeature>> features = null, Optional<string?> description = null, Optional<bool> isPremiumProgressBarEnabled = null, Optional<string> reason = null, CancellationToken ct = default(CancellationToken))

Remarks

Any streams passed to this method will be disposed of at the end of the call. If you want to reuse the streams afterwards, ensure that what you pass is a copy that the method can take ownership of.
